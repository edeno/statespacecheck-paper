% preamble.tex - shared packages and settings for main text and supplement

% Modern encoding (explicit UTF-8 support)
\usepackage[utf8]{inputenc}  % UTF-8 input encoding
\usepackage[T1]{fontenc}     % Modern font encoding (better hyphenation, accents)

% Language support
\usepackage[american]{babel}  % American English hyphenation and typography

% Graphics and figures
\usepackage{graphicx}
\usepackage{subcaption}
\usepackage{float}  % Provides [H] option for exact placement
\usepackage[section]{placeins}  % Prevents floats from crossing section boundaries

% Math packages
\usepackage{amsmath,amssymb,amsthm}

% Page layout
\usepackage[margin=1in]{geometry}

% Better typography
\usepackage{microtype}

% Smart quotations
\usepackage{csquotes}

% Line numbers for review (comment out for final version)
\usepackage{lineno}

% Author affiliations
\usepackage{authblk}

% Professional tables
\usepackage{booktabs}

% Units and numbers
\usepackage{siunitx}

% Bibliography (natbib + BibTeX, matching main.tex). The bibliography
% itself is emitted by \bibliographystyle/\bibliography in the document
% body (see supplement.tex), processed by BibTeX rather than biber.
\usepackage[authoryear]{natbib}

% Hyperlinks (must be near end of preamble)
\usepackage[colorlinks=true,citecolor=blue,linkcolor=blue,urlcolor=blue]{hyperref}

% ORCID support (must come after hyperref)
\usepackage{orcidlink}

% Better cross-references (must come after hyperref)
\usepackage[capitalize,noabbrev]{cleveref}
